Automatically balanced binary tree


                    struct SList {
                        int num;
                        SList *left = nullptr;
                        SList *right = nullptr;
                        unsigned char height;

                        explicit SList(const int num) : left(nullptr), right(nullptr), num(num), height(1) {};
                    };


                    unsigned char height(SList *node) {
                        return node ? node->height : (unsigned char) 0;
                    }

                    void Trueheight(SList *node) {
                        unsigned char hl = height(node->left);
                        unsigned char hr = height(node->right);
                        node->height = (hl > hr ? hl : hr) + (unsigned char) 1;
                    }

                    int bfact(SList *node) {
                        return height(node->right) - height(node->left);
                    }

                    SList *Rrot(SList *node) {
                        SList *nd = node->left;
                        node->left = nd->right;
                        nd->right = node;
                        Trueheight(node);
                        Trueheight(nd);
                        return nd;
                    }

                    SList *Lrot(SList *node) {
                        SList *nd = node->right;
                        node->right = nd->left;
                        nd->left = node;
                        Trueheight(node);
                        Trueheight(nd);
                        return nd;
                    }

                    SList *balance(SList *node) {
                        if (!node)
                            return nullptr;
                        Trueheight(node);
                        if (bfact(node) == 2) {
                            if (bfact(node->right) < 0)
                                node->right = Rrot(node->right);
                            return Lrot(node);
                        } else if (bfact(node) == -2) {
                            if (bfact(node->left) > 0)
                                node->left = Lrot(node->left);
                            return Rrot(node);
                        }
                        return node;
                    }

                    SList *insert(SList *node, int num) {
                        if (!node) return new SList(num);
                        if (node->num < num) {
                            node->right = insert(node->right, num);
                        } else {
                            node->left = insert(node->left, num);
                        }
                        return balance(node);
                    }

                    SList *del(SList *node, int num) {
                        if (!node) return nullptr;
                        if (node->num < num) {
                            node->right = del(node->right, num);
                        } else if (node->num > num) {
                            node->left = del(node->left, num);
                        } else {
                            if (!(node->left || node->right))
                                return nullptr;
                            else if (!node->left)
                                return node->right;
                            else {
                                SList *nd = node->left;
                                while (nd->right)
                                    nd = nd->right;
                                node->num = nd->num;
                                node->left = balance(del(node->left, nd->num));
                            }
                        }
                        return balance(node);
                    }

                    int find(SList *node, int num) {
                        while (node) {
                            if (node->num > num)
                                node = node->left;
                            else if (node->num < num)
                                node = node->right;
                            else
                                return 1;
                        }
                        return 0;

                    }
